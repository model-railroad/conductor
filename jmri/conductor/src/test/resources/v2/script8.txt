# throttles

Throttle CONSIST = 204

# states

Enum PA-State = Init Idle Fwd Rev      # Enum is initialized with its first value.
Enum PA-Motion = Off Idle On
Enum PA-Train = Train1 Train2
Int Speed = 8

# sensors

Sensor B1         = NS48      # 48:1
Sensor B2         = NS49      # 48:2
Sensor PA-TOGGLE  = NS50      # 48:3

# turnouts

Turnout T1        = NT01


# RTAC variables are built-in and do not need to be defined.
# Import Enum RTAC-Motion = Off On
# Export String RTAC-PSA-Text = "Automation Not Working"


# define timers and their default value in seconds

# Maps and Routes info

Int PA-Counter = 0
Int BL-Counter = 0

Route Passenger  = Toggle: PA-TOGGLE, Status: PA-State, Counter: PA-Counter, Throttle: CONSIST
Route Branchline = Toggle: PA-TOGGLE, Status: PA-State, Counter: BL-Counter, Throttle: CONSIST


# GA Tracking

GA-Tracking-Id = "@~/bin/JMRI/rtac_ga_tracking_id.txt"
String GA-URL = "http://consist.alfray.com/train/"


# -------------
# Events Motion
# -------------

Timer PA-PresenceTimer = 300  # 5 minutes in seconds

# PA-Toggle & PA-Motion == On & PA-State == Station -> Train Starts
!PA-Toggle -> PA-Motion = Off
PA-Toggle & PA-Motion == On   & RTAC-Motion == Off -> PA-Motion = Idle
PA-Toggle & PA-Motion == Idle & RTAC-Motion == On  -> PA-Motion = On ; PA-PresenceTimer Start
PA-Toggle & PA-Motion == On   & PA-PresenceTimer -> PA-Motion = Idle

PA-Motion == Off  -> RTAC-PSA-Text = "Train Automation Not Working"
PA-Motion == Idle -> RTAC-PSA-Text = "Train Automation Idle"
PA-Motion == On   -> RTAC-PSA-Text = "Train is Running"
PA-State == Idle & !PA-Motion == Off -> RTAC-PSA-Text = "Train Stopped.\nNext train starts in {now+10m} minutes"


# ---------
# Events PA
# ---------

# Setup
PA-State == INIT -> PA-State = Idle; CONSIST Sound = 1; CONSIST Light = 1

PA-State == Idle & PA-Motion == On ->
    PA-State = Fwd ;
    CONSIST Horn ;
    CONSIST Forward = Speed ;
    PA-Counter += 1
PA-State == Fwd  & B1 + 5 & CONSIST Forward -> PA-State = Rev; CONSIST Reverse = Speed;
PA-State == Rev  & B1     & CONSIST Reverse -> CONSIST Stop; PA-State = Idle

PA-State != Idle -> T1 Normal;

PA-Toggle  -> GA-Event Category: Automation, Action: On,  Label: Passenger, User: Staff
!PA-Toggle -> GA-Event Category: Automation, Action: Off, Label: Passenger, User: Staff

PA-State == Fwd -> GA-Page URL: GA-URL, Path: PA-Train, User: PA-Counter ;
                GA-Event Category: Activation, Action: Start, Label: PA-Train, User: PA-Counter
PA-State == Rev -> GA-Event Category: Activation, Action: Stop , Label: PA-Train, User: PA-Counter
