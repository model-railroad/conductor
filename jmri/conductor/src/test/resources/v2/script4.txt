# throttles

Throttle CONSIST = 204 209
Throttle FRONT = 204
Throttle REAR  = 209

# states

Int Init  = 1
Int Speed = 8


# sensors

Sensor B1         = NS48      # 48:1
Sensor PA-TOGGLE  = NS49      # 48:2

# turnouts


# define timers and their default value in seconds


# ---------
# Events PA
# ---------

# Setup
INIT -> FRONT Sound = 1; FRONT Light = 1; REAR Sound = 1; REAR Light = 1

PA-TOGGLE -> FRONT Horn ; CONSIST Forward = Speed
B1 + 5 & CONSIST Forward -> CONSIST Reverse = Speed ;
B1     & CONSIST Reverse -> CONSIST Stop
