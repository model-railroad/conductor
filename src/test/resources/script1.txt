# Comment
Throttle T1 = 1201
Throttle T2 = 1202
Var Var-1       = 1
Var Half-Speed  = 4
Var Full-Speed  = 8

# define alias for sensors
Sensor Alias   = NS784

# define timers and their default value in seconds
Timer Timer-1 = 5

# events
Alias + Var-1 -> T1 Forward = Half-Speed ; Var-1 = 0 ; T1 Sound = 1; T1 Light = 1 ; T1 Horn
Alias + !Var-1 -> T1 Forward = Full-Speed ; Start = Timer-1
T1 Forward -> T1 Forward = Full-Speed
T1 Reverse -> T1 Reverse = Half-Speed
T1 Stopped -> T1 Stop
Timer-1 -> End = Timer-1 ; T1 Sound = 0; T1 Horn ; Var-1 = 1 ; T1 Reverse = Half-Speed
!Timer-1 -> T1 Stop
